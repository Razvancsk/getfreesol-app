3. Jupiter API Integration (server/jupiterApiFixed.ts)
This is the actual Jupiter API call:

async getQuote(inputMint, outputMint, amount, slippageBps, platformFeeOptions) {
  const params = new URLSearchParams({
    inputMint,
    outputMint,
    amount: amount.toString(),
    slippageBps: slippageBps.toString()
  });
  // Add referral fee parameters
  if (platformFeeOptions) {
    params.append('platformFeeBps', platformFeeOptions.platformFeeBps.toString());
    params.append('feeAccount', platformFeeOptions.feeAccount);
  }
  const response = await fetch(`https://lite-api.jup.ag/swap/v1/quote?${params}`);
  return response.json();
}
async createSwap(quoteResponse, userPublicKey, options) {
  const payload = {
    quoteResponse: quoteResponse,
    userPublicKey: userPublicKey,
    wrapAndUnwrapSol: true,
    dynamicComputeUnitLimit: true,
    prioritizationFeeLamports: 'auto'
  };
  // Add fee account to swap transaction
  if (options.feeAccount) {
    payload.feeAccount = options.feeAccount;
  }
  const response = await fetch('https://lite-api.jup.ag/swap/v1/swap', {
    method: 'POST',
    body: JSON.stringify(payload)
  });
  
  return response.json();
}
